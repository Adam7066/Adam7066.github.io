Python-02：多線程-加速圖片下載

Aug 03, 2020

簡介

  • 接續上一篇Python-01：爬蟲-圖片下載的內容，這篇文章將帶領大家使用多線程來加速圖片的下載

教學開始

  • 首先我們先看個簡單的多線程範例
 1import time, threading
 2
 3# 子執行緒的工作函數
 4def job():
 5  for i in range(5):
 6    print("Child thread:", i)
 7    time.sleep(1)
 8# 建立一個子執行緒
 9t = threading.Thread(target = job)
10# 執行該子執行緒
11t.start()
12# 主執行緒繼續執行自己的工作
13for i in range(3):
14  print("Main thread:", i)
15  time.sleep(1)
16# 等待 t 這個子執行緒結束
17t.join()
18print("Done.")

 1import time, threading
 2
 3# 子執行緒的工作函數
 4def job(num):
 5  print("Thread", num)
 6  time.sleep(1)
 7# 建立 5 個子執行緒
 8threads = []
 9for i in range(5):
10  threads.append(threading.Thread(target = job, args = (i,)))
11  threads[i].start()
12# 主執行緒繼續執行自己的工作
13
14# 等待所有子執行緒結束
15for i in range(5):
16  threads[i].join()
17print("Done.")

完整程式碼

 1# -*- coding: UTF-8 -*-
 2from selenium import webdriver
 3from bs4 import BeautifulSoup
 4import requests, threading
 5
 6def download_img(img, num):
 7    r = requests.get(img)
 8    with open(save_url + str(num+1) + '.jpg', 'wb') as f:
 9        f.write(r.content)
10    
11if __name__ == "__main__":  
12    save_url = './test/'
13    url = 'https://www.ohmanhua.com/13621/1/1.html'
14    chop = webdriver.ChromeOptions()
15    chop.add_extension('Adblock-Plus_v3.8.4.crx')
16    browser = webdriver.Chrome(options = chop)
17    browser.implicitly_wait(10)
18    browser.get(url)
19    soup = BeautifulSoup(browser.page_source, 'lxml')
20    img_count = int(soup.find('select', {'class': 'mh_select'}).find_all('option')[-1].get('value'))
21    img_url = soup.find_all('div', {'class': 'mh_comicpic'})[0].find('img').get('src')
22    if img_url[0] == '/':
23        img_url = 'https:' + img_url
24    m = len(img_url.rsplit('/', 1)[1].split('.')[0])
25    img_url = img_url.rsplit('/', 1)[0] + '/'
26    th_list = []
27    for i in range(img_count):
28        img = img_url + str(i + 1).zfill(m) + '.jpg'
29        th = threading.Thread(target=download_img, args=(img, i))
30        th.start()
31        th_list.append(th)
32    for th in th_list:
33        th.join()
34    browser.quit()
