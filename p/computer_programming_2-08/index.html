 11/*
 12 * Insert a new entry between two known consecutive entries.
 13 *
 14 * This is only for internal list manipulation where we know
 15 * the prev/next entries already!
 16 */
 17static inline void __list_add(struct list_head *new,
 18                              struct list_head *prev,
 19                              struct list_head *next) {
 20    next->prev = new;
 21    new->next = next;
 22    new->prev = prev;
 23    prev->next = new;
 24}

 26/**
 27 * list_add - add a new entry
 28 * @new: new entry to be added
 29 * @head: list head to add it after
 30 *
 31 * Insert a new entry after the specified head.
 32 * This is good for implementing stacks.
 33 */
 34static inline void list_add(struct list_head *new, struct list_head *head) {
 35    __list_add(new, head, head->next);
 36}

 39/**
 40 * list_add_tail - add a new entry
 41 * @new: new entry to be added
 42 * @head: list head to add it before
 43 *
 44 * Insert a new entry before the specified head.
 45 * This is useful for implementing queues.
 46 */
 47static inline void list_add_tail(struct list_head *new, struct list_head *head) {
 48    __list_add(new, head->prev, head);

 51/*
 52 * Delete a list entry by making the prev/next entries
 53 * point to each other.
 54 *
 55 * This is only for internal list manipulation where we know
 56 * the prev/next entries already!
 57 */
 58static inline void __list_del(struct list_head * prev, struct list_head * next) {
 59    next->prev = prev;
 60    prev->next = next;
 61}
 62
 63/**
 64 * list_del - deletes entry from list.
 65 * @entry: the element to delete from the list.
 66 * Note: list_empty() on entry does not return true after this, the entry is
 67 * in an undefined state.
 68 */
 69static inline void __list_del_entry(struct list_head *entry) {
 70    if (entry == NULL)
 71        return;
 72
 73    __list_del(entry->prev, entry->next);
 74}
 75
 76static inline void list_del(struct list_head *entry) {
 77    __list_del_entry(entry);
 78    entry->next = NULL;
 79    entry->prev = NULL;
 80}
