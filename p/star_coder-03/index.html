B - Matrix Chain Multiplication

 1#include <bits/stdc++.h>
 2using namespace std;
 3struct Matrix {
 4    int row, col;
 5};
 6int main() {
 7    int n, errCnt = -1;
 8    cin >> n;
 9    map<char, Matrix> ma;
10    for(int i = 0; i < n; ++i) {
11        char name;
12        Matrix m;
13        cin >> name >> m.row >> m.col;
14        ma[name] = m;
15    }
16    string s;
17    while(cin >> s) {
18        int multiCnt = 0;
19        stack<Matrix> st;
20        for(int i = 0; i < s.length(); ++i) {
21            if(s[i] == '(') continue;
22            if(s[i] != ')') {
23                st.push(ma[s[i]]);
24                continue;
25            }
26            Matrix b = st.top();
27            st.pop();
28            Matrix a = st.top();
29            st.pop();
30            if(a.col != b.row) {
31                multiCnt = errCnt;
32                break;
33            }
34            multiCnt += a.row * b.row * b.col;
35            st.push({a.row, b.col});
36        }
37        if(multiCnt == errCnt) cout << "error\n";
38        else cout << multiCnt << endl;
39    }
40    return 0;
41}

C - Printer Queue

 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    int T;
 5    cin >> T;
 6    while(T--) {
 7        int n, m;
 8        cin >> n >> m;
 9        deque<pair<int, int>> dq;
10        int t[10] = {0};
11        for(int i = 0; i < n; ++i) {
12            int a;
13            cin >> a;
14            dq.emplace_back(a, i);
15            ++t[a];
16        }
17        int cnt = 0;
18        int lev, idx;
19        while(true) {
20            tie(lev, idx) = dq.front();
21            int canPop = 1;
22            for(int i = lev + 1; i < 10; ++i) canPop &= (t[i] == 0);
23            if(canPop) {
24                ++cnt;
25                --t[lev];
26                if(idx == m) break;
27                dq.pop_front();
28            }
29            else {
30                dq.push_back(dq.front());
31                dq.pop_front();
32            }
33        }
34        cout << cnt << '\n';
35    }
36    return 0;
37}

D - Uncompress

 1#include <bits/stdc++.h>
 2using namespace std;
 3int main() {
 4    vector<string> vs;
 5    string buf;
 6    while (getline(cin, buf) && buf[0] != '0') {
 7        string s;
 8        for (int i = 0; buf[i]; ++ i) {
 9            if (isalpha(buf[i])) {
10                s = "";
11                while (isalpha(buf[i]))
12                    s.insert(s.end(), buf[i ++]);
13                vs.push_back(s);
14                --i;
15                cout << s;
16            }
17            else if (isdigit(buf[i])) {
18                int value = 0;
19                while (isdigit(buf[i]))
20                    value = value * 10 + buf[i ++] - '0';
21                s = vs[vs.size() - value];
22                vs.erase(vs.end() - value);
23                vs.push_back(s);
24                --i;
25                cout << s;
26            }
27            else printf("%c", buf[i]);
28        }
29        puts("");
30    }
31    return 0;
32}

18                qu.push(b);

32                    else flag[2] = false;

38        else if(flag[0]) cout<<"stack\n";
39        else if(flag[1]) cout<<"queue\n";
40        else if(flag[2]) cout<<"priority queue\n";

G - Ubiquitous Religions

 1#include <bits/stdc++.h>
 2#define USE_CPPIO() ios_base::sync_with_stdio(0); cin.tie(0)
 3using namespace std;
 4int dsu[50005];
 5void init(int n) {
 6    for(int i = 0; i < n; ++i)
 7        dsu[i] = i;

 9int Find(int x) {

H - Almost Union-Find

 1#include <bits/stdc++.h>
 2#define USE_CPPIO() ios_base::sync_with_stdio(0); cin.tie(0)
 3using namespace std;
 4int dsu[200005], num[200005], sum[200005];
 5void init(int n) {
 6    for(int i = 0; i <= n; ++i) {
 7        dsu[i] = dsu[i + n] = i + n;
 8        sum[i] = sum[i + n] = i;
 9        num[i] = num[i + n] = 1;
10    }
11}
12int Find(int x) {
13    if(x == dsu[x]) return x;
14    return dsu[x] = Find(dsu[x]);
15}
16int main() {
17    USE_CPPIO();
18    int n, m;
19    while(cin >> n >> m) {
20        init(n);
21        for(int i = 0; i < m; ++i) {
22            int t, a, b;
23            cin >> t >> a;
24            if(t == 1) {
25                cin >> b;
26                int x = Find(a);
27                int y = Find(b);
28                if(x == y) continue;
29                dsu[x] = y;
30                num[y] += num[x];
31                sum[y] += sum[x];
32            }
33            else if(t == 2) {
34                cin >> b;
35                int x = Find(a);
36                int y = Find(b);
37                if(x == y) continue;
38                dsu[a] = y;
39                --num[x];
40                ++num[y];
41                sum[x] -= a;
42                sum[y] += a;
43            }
44            else if(t == 3) {
45                int x = Find(a);
46                cout << num[x] << " " << sum[x] << "\n";
47            }
48        }
49    }
50    return 0;
51}

I - Islands

 1#include <bits/stdc++.h>
 2#define USE_CPPIO() ios_base::sync_with_stdio(0); cin.tie(0)
 3using namespace std;
 4int n, m, total, dsu[1000010];
 5int dx[4] = {0, 0, 1, -1};
 6int dy[4] = {1, -1, 0, 0};
 7struct Point {
 8    int x, y, id, height;
 9};
10vector<Point> vp;
11bool cmp(Point a, Point b) {
12    return a.height > b.height;
13}
14int Find(int t) {
15    if(t == -1) return -1;
16    if(dsu[t] == t) return t;
17    return dsu[t] = Find(dsu[t]);
18}
19void add(int idx) {
20    int id = vp[idx].id;
21    dsu[id] = id;
22    ++total;
23    for(int i = 0; i < 4; ++i) {
24        int x = vp[idx].x + dx[i];
25        int y = vp[idx].y + dy[i];
26        if(x >= 0 && x < n && y >= 0 && y < m) {
27            int id2 = x * m + y;
28            int root_id2 = Find(id2);
29            if(root_id2 == -1) continue;
30            if(root_id2 != id) {
31                --total;
32                dsu[root_id2] = id;
33            }
34        }
35    }
36}
37int main() {
38    USE_CPPIO();
39    int T;
40    cin >> T;
41    while(T--) {
42        cin >> n >> m;
43        vp.clear();
44        vector<int> h;
45        for(int i = 0; i < n; ++i) {
46            for(int j = 0; j < m; ++j) {
47                int t;
48                cin >> t;
49                Point p = {i, j, i * m + j, t};
50                vp.push_back(p);
51                dsu[i * m + j] = -1;
52            }
53        }
54        sort(vp.begin(), vp.end(), cmp);
55        int q;
56        cin >> q;
57        for(int i = 0; i < q; ++i) {
58            int t;
59            cin >> t;
60            h.push_back(t);
61        }
62        int idx = 0;
63        total = 0;
64        for(int i = h.size() - 1; i >= 0; --i) {
65            while(idx < vp.size() && h[i] < vp[idx].height)
66                add(idx++);
67            h[i] = total;
68        }
69        for(auto i : h) cout << i << " ";
70        cout << endl;
71    }
72    return 0;
73}

J - Borrowers

 1#include <bits/stdc++.h>
 2#define USE_CPPIO() ios_base::sync_with_stdio(0); cin.tie(0)
 3using namespace std;
 4map<string, string> mss;
 5struct ReturnBook {
 6    string title;
 7    string author;
 8};
 9bool cmp(string b1, string b2) {
10    string a1 = mss[b1];
11    string a2 = mss[b2];
12    if(a1 != a2) return a1 < a2;
13    return b1 < b2;
14}
15bool cmp_return(ReturnBook b1, ReturnBook b2) {
16    if(b1.author != b2.author) return b1.author < b2.author;
17    return b1.title < b2.title;
18}
19int main() {
20    USE_CPPIO();
21    string s;
22    vector<string> shelve;
23    vector<ReturnBook> vRB;
24    while(getline(cin, s)) {
25        if(s == "END") break;
26        int idx = s.find("\" by");
27        string title = s.substr(0, idx + 1);
28        string author = s.substr(idx + 4);
29        mss[title] = author;
30        shelve.push_back(title);
31    }
32    sort(shelve.begin(), shelve.end(), cmp);
33    while(getline(cin, s)) {
34        if(s == "END") break;
35        if(s == "SHELVE") {
36            sort(vRB.begin(), vRB.end(), cmp_return);
37            for(int i = 0; i < vRB.size(); ++i) {
38                auto it = lower_bound(shelve.begin(), shelve.end(), vRB[i].title, cmp);
39                if(it == shelve.begin()) cout << "Put " << vRB[i].title << " first" << endl;
40                else {
41                    int t = it - shelve.begin() - 1;
42                    cout << "Put " << vRB[i].title << " after " << shelve[t] << endl;
43                }
44                shelve.insert(it, vRB[i].title);
45            }
46            vRB.clear();
47            cout << "END\n";
48        }
49        else {
50            int idx = s.find("\"");
51            string title = s.substr(idx);
52            if(s[0] == 'B') {
53                auto it = lower_bound(shelve.begin(), shelve.end(), title, cmp);
54                shelve.erase(it);
55            }
56            else vRB.push_back({title, mss[title]});
57        }
58    }
59    return 0;
60}

K - Database

 1#include <bits/stdc++.h>
 2#define USE_CPPIO() ios_base::sync_with_stdio(0); cin.tie(0)
 3using namespace std;
 4map<string, int> ID;
 5int idx = 1;
 6int StrID(string str) {
 7    if(ID[str]) return ID[str];
 8    return ID[str] = idx++;
 9}
10int main() {
11    USE_CPPIO();
12    int row, col;
13    while(cin >> row >> col) {
14        cin.get();
15        int IDtable[row + 2][col + 2];
16        string table[row + 2][col + 2], tmp;
17        for(int i = 1; i <= row; ++i) {
18            getline(cin, tmp);
19            char *pch = strtok((char*)tmp.c_str(), ",");
20            int j = 1;
21            while(pch) {
22                table[i][j++] = pch;
23                pch = strtok(NULL, ",");
24            }
25        }
26        for(int i = 1; i <= row; ++i) {
27            for(int j = 1; j <= col; ++j) {
28                IDtable[i][j] = StrID(table[i][j]);
29            }
30        }
31        bool flag = true;
32        for(int i = 1; i <= col && flag; ++i) {
33            for(int j = i + 1; j <= col && flag; ++j) {
34                map<int, int> mii;
35                for(int k = 1; k <= row && flag; ++k) {
36                    int idx = IDtable[k][i] * 100000 + IDtable[k][j];
37                    if(mii.count(idx)) {
38                        cout << "NO" << endl;
39                        cout << mii[idx] << " " << k << endl;
40                        cout << i << " " << j << endl;
41                        flag = false;
42                    }
43                    else mii[idx] = k;
44                }
45            }
46        }
47        if(flag) cout << "YES" << endl;
48    }
49    return 0;
50}
